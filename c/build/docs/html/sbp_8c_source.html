161 s8 sbp_register_callback(sbp_state_t *s, u16 msg_type, sbp_msg_callback_t cb, void *context,
162  sbp_msg_callbacks_node_t *node)
163 {
164  /* Check our callback function pointer isn't NULL. */
165  if (cb == 0)
166  return SBP_NULL_ERROR;
167 
168  /* Check our callback node pointer isn't NULL. */
169  if (node == 0)
170  return SBP_NULL_ERROR;
171 
172  /* Check if callback was already registered for this type. */
173  if (sbp_find_callback(s, msg_type) != 0)
174  return SBP_CALLBACK_ERROR;
175 
176  /* Fill in our new sbp_msg_callback_node_t. */
177  node->msg_type = msg_type;
178  node->cb = cb;
179  node->context = context;
180  /* The next pointer is set to NULL, i.e. this
181  * will be the new end of the linked list.
182  */
183  node->next = 0;
184 
185  /* If our linked list is empty then just
186  * add the new node to the start.
187  */
188  if (s->sbp_msg_callbacks_head == 0) {
189  s->sbp_msg_callbacks_head = node;
190  return SBP_OK;
191  }
192 
193  /* Find the tail of our linked list and
194  * add our new node to the end.
195  */
196  sbp_msg_callbacks_node_t *p = s->sbp_msg_callbacks_head;
197  while (p->next)
198  p = p->next;
199 
200  p->next = node;
201 
202  return SBP_OK;
203 }

222 sbp_msg_callbacks_node_t* sbp_find_callback(sbp_state_t *s, u16 msg_type)
223 {
224  /* If our list is empty, return NULL. */
225  if (!s->sbp_msg_callbacks_head)
226  return 0;
227 
228  /* Traverse the linked list and return the callback
229  * function pointer if we find a node with a matching
230  * message id.
231  */
232  sbp_msg_callbacks_node_t *p = s->sbp_msg_callbacks_head;
233  do
234  if (p->msg_type == msg_type)
235  return p;
236 
237  while ((p = p->next));
238 
239  /* Didn't find a matching callback, return NULL. */
240  return 0;
241 }

312 s8 sbp_process(sbp_state_t *s, u32 (*read)(u8 *buff, u32 n, void *context))
313 {
314  u8 temp;
315  u16 crc;
316 
317  switch (s->state) {
318  case WAITING:
319  if ((*read)(&temp, 1, s->io_context) == 1)
320  if (temp == SBP_PREAMBLE) {
321  s->n_read = 0;
322  s->state = GET_TYPE;
323  }
324  break;
325 
326  case GET_TYPE:
327  s->n_read += (*read)((u8*)&(s->msg_type) + s->n_read,
328  2-s->n_read, s->io_context);
329  if (s->n_read >= 2) {
330  /* Swap bytes to little endian. */
331  s->n_read = 0;
332  s->state = GET_SENDER;
333  }
334  break;
335 
336  case GET_SENDER:
337  s->n_read += (*read)((u8*)&(s->sender_id) + s->n_read,
338  2-s->n_read, s->io_context);
339  if (s->n_read >= 2) {
340  /* Swap bytes to little endian. */
341  s->state = GET_LEN;
342  }
343  break;
344 
345  case GET_LEN:
346  if ((*read)(&(s->msg_len), 1, s->io_context) == 1) {
347  s->n_read = 0;
348  s->state = GET_MSG;
349  }
350  break;
351 
352  case GET_MSG:
353  /* Not received whole message yet, try and read some more. */
354  s->n_read += (*read)(
355  &(s->msg_buff[s->n_read]),
356  s->msg_len - s->n_read,
357  s->io_context
358  );
359  if (s->msg_len - s->n_read <= 0) {
360  s->n_read = 0;
361  s->state = GET_CRC;
362  }
363  break;
364 
365  case GET_CRC:
366  s->n_read += (*read)((u8*)&(s->crc) + s->n_read,
367  2-s->n_read, s->io_context);
368  if (s->n_read >= 2) {
369  s->state = WAITING;
370 
371  /* Swap bytes to little endian. */
372  crc = crc16_ccitt((u8*)&(s->msg_type), 2, 0);
373  crc = crc16_ccitt((u8*)&(s->sender_id), 2, crc);
374  crc = crc16_ccitt(&(s->msg_len), 1, crc);
375  crc = crc16_ccitt(s->msg_buff, s->msg_len, crc);
376  if (s->crc == crc) {
377 
378  /* Message complete, process it. */
379  sbp_msg_callbacks_node_t* node = sbp_find_callback(s, s->msg_type);
380  if (node) {
381  (*node->cb)(s->sender_id, s->msg_len, s->msg_buff, node->context);
382  return SBP_OK_CALLBACK_EXECUTED;
383  } else {
384  return SBP_OK_CALLBACK_UNDEFINED;
385  }
386  } else
387  return SBP_CRC_ERROR;
388  }
389  break;
390 
391  default:
392  s->state = WAITING;
393  break;
394  }
395 
396  return SBP_OK;
397 }

428 s8 sbp_send_message(sbp_state_t *s, u16 msg_type, u16 sender_id, u8 len, u8 *payload,
429  u32 (*write)(u8 *buff, u32 n, void *context))
430 {
431  /* Check our payload data pointer isn't NULL unless len = 0. */
432  if (len != 0 && payload == 0)
433  return SBP_NULL_ERROR;
434 
435  /* Check our write function pointer isn't NULL. */
436  if (write == 0)
437  return SBP_NULL_ERROR;
438 
439  u16 crc;
440 
441  u8 preamble = SBP_PREAMBLE;
442  if ((*write)(&preamble, 1, s->io_context) != 1)
443  return SBP_SEND_ERROR;
444 
445  if ((*write)((u8*)&msg_type, 2, s->io_context) != 2)
446  return SBP_SEND_ERROR;
447 
448  if ((*write)((u8*)&sender_id, 2, s->io_context) != 2)
449  return SBP_SEND_ERROR;
450 
451  if ((*write)(&len, 1, s->io_context) != 1)
452  return SBP_SEND_ERROR;
453 
454  if (len > 0) {
455  if ((*write)(payload, len, s->io_context) != len)
456  return SBP_SEND_ERROR;
457  }
458 
459  crc = crc16_ccitt((u8*)&(msg_type), 2, 0);
460  crc = crc16_ccitt((u8*)&(sender_id), 2, crc);
461  crc = crc16_ccitt(&(len), 1, crc);
462  crc = crc16_ccitt(payload, len, crc);
463 
464  if ((*write)((u8*)&crc, 2, s->io_context) != 2)
465  return SBP_SEND_ERROR;
466 
467  return SBP_OK;
468 }
